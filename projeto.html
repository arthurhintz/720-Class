D = n * k
block <- 4

# 1. Coordinates for every plot in the field
coords <- expand.grid(row = 1:n, col = 1:k)

# 2. Define the Gaussian semivariogram model
#    - nugget : measurement noise
#    - psill  : variance
#    - range  : distance at which spatial correlation becomes negligible

vmodel <- vgm(psill = s^2, model = "Gau", range = 5, nugget = 1)

# 3. Set up an unconditional simulation generator 
g_dummy <- gstat(formula = z ~ 1, locations = ~row + col, dummy = TRUE,
                  beta = 0, model = vmodel, nmax = 20)

# 4. Draw one realization on your grid
set.seed(1)
sim <- predict(g_dummy, newdata = coords, nsim = 1)
[using unconditional Gaussian simulation]
# 5. Reshape the simulated spatial errors back into your n x k field
spatial_error <- matrix(sim$sim1, nrow = n, ncol = k)

# 6. Rebuild your field: trend (runif) + block effect (rbinom) + spatial error
field <- matrix(NA, nrow = n, ncol = k)
for (i in 1:n) {
  field[i, ] <- runif(k) + rbinom(k, block, 0.5)
}
field <- field + spatial_error

image(field)
